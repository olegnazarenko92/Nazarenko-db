# Домашнее задание к занятию «ELK»

### Инструкция по выполнению домашнего задания
version: '3.7'

services:
  elasticsearch:
    image: elasticsearch:7.17.10
    environment:
      - discovery.type=single-node
      - cluster.name=netology-cluster
    ports:
      - "9200:9200"
    networks:
      - elk

  kibana:
    image: kibana:7.17.10
    ports:
      - "5601:5601"
    depends_on:
      - elasticsearch
    networks:
      - elk

networks:
  elk:
    driver: bridge
---
